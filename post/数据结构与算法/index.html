# 递归
def inorder(root):
    if root:
        inorder(root.left)
        print(root.val)
        inorder(root.right)
# 迭代
stack = []
while root or stack:
    while root:
        stack.append(root)
        root = root.left
    root = stack.pop()
    print(root.val)
    root = root.right
